1、配置奖项：使用记事本打开config.js这个文件，然后按照里面的示例，修改成自己想要的即可，如下：

2、配置 awardsType 奖项类型，1 表示不分等级，不限制数量，2 表示分等级，限制数量，
	1）如果awardsType=1 配置下面的奖项即可：
	var globalkeys = [
		"999元",
		"试卷一套",
		"辣条一包",
		"放假一天"
		...  这使用英文的双引号包含新添加的内容即可，最后使用英文逗号结束
	];


	2）如果awardsType=2 配置下面的奖项即可：
	var globalkeys2 = {
	"一等奖":{ //这里是奖项等级描述
			"awards":"888元", //这里是具体的奖项内容
			"num":3 //这个是每个奖项的数量
			},
	"二等奖":{
			"awards":"试卷一套", //这里是具体的奖项内容
			"num":2 //这个是每个奖项的数量
			},
	"三等奖":{
			"awards":"辣条一包", //这里是具体的奖项内容
			"num":1  //这个是每个奖项的数量
			}
		.... 这个可以设置其他等级的奖项，但是格式要求和上面一致
	};

2、双击index.html 文件，或者鼠标右键使用浏览器打开即可抽奖

3、每次抽奖完成后，点击重新抽奖按钮，即可重新随机抽取自定义奖项
